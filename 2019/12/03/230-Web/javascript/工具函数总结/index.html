// 获取URL地址参数数组
function getQueryFromURL(url) {
    url = url || 'http://qq.com/s?a=b#rd'; 
    var tmp = url.split('?'),
        paramStr = (tmp[1] || "").split('#')[0],
        query = paramStr ? paramStr.split('&') : [],
        params = {};
    for (var i=0; i<query.length; i++) {
        var arg = query[i].split('=');
        params[arg[0]] = arg[1];
    }
    return params;
};
